00062 template<typename T> const T& SearchAndReplace(T& text, const T& pattern, const T& replace)
00063 {
00064         T replacement;
00065         if ((!pattern.empty()) && (!text.empty()))
00066         {
00067                 for (std::string::size_type n = 0; n != text.length(); ++n)
00068                 {
00069                         if (text.length() >= pattern.length() && text.substr(n, pattern.length()) == pattern)
00070                         {
00071                                 /* Found the pattern in the text, replace it, and advance */
00072                                 replacement.append(replace);
00073                                 n = n + pattern.length() - 1;
00074                         }
00075                         else
00076                         {
00077                                 replacement += text[n];
00078                         }
00079                 }
00080         }
00081         text = replacement;
00082         return text;
00083 }
