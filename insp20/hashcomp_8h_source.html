00063 template<typename T> const T& SearchAndReplace(T& text, const T& pattern, const T& replace)
00064 {
00065         T replacement;
00066         if ((!pattern.empty()) && (!text.empty()))
00067         {
00068                 for (std::string::size_type n = 0; n != text.length(); ++n)
00069                 {
00070                         if (text.length() >= pattern.length() && text.substr(n, pattern.length()) == pattern)
00071                         {
00072                                 /* Found the pattern in the text, replace it, and advance */
00073                                 replacement.append(replace);
00074                                 n = n + pattern.length() - 1;
00075                         }
00076                         else
00077                         {
00078                                 replacement += text[n];
00079                         }
00080                 }
00081         }
00082         text = replacement;
00083         return text;
00084 }
