00108 template<typename T> inline std::string ConvNumeric(const T &in)
00109 {
00110         if (in == 0) return "0";
00111         char res[MAXBUF];
00112         char* out = res;
00113         T quotient = in;
00114         while (quotient) {
00115                 *out = "0123456789"[ std::abs( (long)quotient % 10 ) ];
00116                 ++out;
00117                 quotient /= 10;
00118         }
00119         if (in < 0)
00120                 *out++ = '-';
00121         *out = 0;
00122         std::reverse(res,out);
00123         return res;
00124 }
